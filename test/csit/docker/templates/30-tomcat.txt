# Set up tomcat
RUN wget -q http://mirrors.ocf.berkeley.edu/apache/tomcat/tomcat-8/v8.5.9/bin/apache-tomcat-8.5.9.tar.gz && tar --strip-components=1 -xf apache-tomcat-8.5.9.tar.gz && rm -f apache-tomcat-8.5.9.tar.gz && rm -rf webapps && mkdir -p webapps/ROOT
ENV CATALINA_HOME /service
